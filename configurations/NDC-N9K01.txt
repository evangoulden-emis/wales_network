feature interface-vlan

  ip ospf 1 area 0
  ip ospf network point-to-point

  description "dhcw-ndc-dxcon-01" ! # Add DXCON ID
  switchport mode trunk
  switchport trunk allowed vlan 101,201
  bfd interval 50 min_rx 50 multiplier 5
  no shutdown

vrf context FE
vrf context BE


vrf definition 
  description ""


vlan 101
  name FrontEnd-VLAN-101
vlan 201
  name BackEnd-VLAN-201

  no shutdown

  description "Primary frontend VLAN for AWS Connectivity"
  ip address 169.254.11.1 255.255.255.252
  no shutdown

  bgp log-neighbor-changes

  timers bgp 10 30
  maximum-paths 4
  vrf FE
    neighbor 169.254.255.2
      bfd
      remote-as 64521
      update-source loopback100
      address-family ipv4 unicast
    neighbor 169.254.1.2
      bfd
